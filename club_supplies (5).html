部署 Apps Script（一次性設定）
1
開啟你的 Google 試算表，確認有兩個工作表分別命名為 「耗材庫存」 和 「出入庫記錄」
2
點上方選單 擴充功能 → Apps Script，刪除舊內容，貼上下方程式碼，並把 SS_ID 換成你的試算表 ID
3
如何找到試算表 ID：開啟 Google 試算表，網址列長這樣：
https://docs.google.com/spreadsheets/d/【這段就是ID】/edit
4
點 部署 → 新增部署 → 類型選「網頁應用程式」
執行身分：我　　存取權：所有人（含匿名）
5
複製部署後的網址，貼到上方「Apps Script Web App 網址」欄位，按測試連線確認成功後，按同步到 Sheets
6
下次接手的社員只需填入網址，按「從 Sheets 載入」就能取回所有資料 ✓
const SS_ID = '你的試算表ID'; // ← 換成你的ID const SUPPLY_SHEET = '耗材庫存'; const LOG_SHEET = '出入庫記錄'; const PHOTO_FOLDER = '耗材照片'; // GET：讀取資料 function doGet(e) { const action = e.parameter.action; if (action === 'ping') return ok({status: 'ok'}); if (action === 'getAll') return ok(getAllData()); return ok({error: 'unknown action'}); } // POST：寫入資料（與 booking admin 相同方式） function doPost(e) { try { const body = JSON.parse(e.postData.contents); if (body.action === 'saveAll') { saveAll(body.data); return ok({saved: true}); } return ok({error: 'unknown action'}); } catch(err) { return ok({error: err.message}); } } function getAllData() { const ss = SpreadsheetApp.openById(SS_ID); return { supplies: sheetToJson(ss.getSheetByName(SUPPLY_SHEET)), logs: sheetToJson(ss.getSheetByName(LOG_SHEET)) }; } function saveAll(data) { const ss = SpreadsheetApp.openById(SS_ID); // 圖片：base64 上傳到 Drive，改存公開連結 const folder = getPhotoFolder(); if (data.supplies) { data.supplies = data.supplies.map(function(s) { if (s.photo && s.photo.indexOf('data:image') === 0) { try { const parts = s.photo.split(','); const meta = parts[0]; // e.g. data:image/png;base64 const b64 = parts[1]; const mime = meta.split(':')[1].split(';')[0]; const ext = mime.split('/')[1]; const blob = Utilities.newBlob(Utilities.base64Decode(b64), mime, s.name + '_' + s.id + '.' + ext); const existing = folder.getFilesByName(blob.getName()); while (existing.hasNext()) existing.next().setTrashed(true); const file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); s.photo = 'https://drive.google.com/thumbnail?id=' + file.getId() + '&sz=w400'; } catch(err) { s.photo = ''; } } return s; }); } jsonToSheet(ss.getSheetByName(SUPPLY_SHEET), data.supplies); jsonToSheet(ss.getSheetByName(LOG_SHEET), data.logs); } function getPhotoFolder() { const folders = DriveApp.getFoldersByName(PHOTO_FOLDER); if (folders.hasNext()) return folders.next(); return DriveApp.createFolder(PHOTO_FOLDER); } function sheetToJson(sheet) { if (!sheet) return []; const vals = sheet.getDataRange().getValues(); if (vals.length < 2) return []; const hdr = vals[0]; return vals.slice(1).map(function(r) { return Object.fromEntries(hdr.map(function(h, i) { return [h, r[i]]; })); }); } function jsonToSheet(sheet, arr) { if (!sheet || !arr || !arr.length) return; const hdr = Object.keys(arr[0]); sheet.clearContents(); sheet.appendRow(hdr); arr.forEach(function(r) { sheet.appendRow(hdr.map(function(h) { return r[h] !== undefined ? r[h] : ''; })); }); } function ok(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
