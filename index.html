1
開啟SPTT賬號的 Google 試算表，確認有兩個工作表分別命名為 「耗材庫存」 和 「出入庫記錄」
2
點上方選單 擴充功能 → Apps Script，刪除舊內容，貼上下方程式碼，並把 SS_ID 換成試算表 ID
3
如何找到試算表 ID：開啟 Google 試算表，網址列長這樣：
https://docs.google.com/spreadsheets/d/【這段就是ID】/edit
4
點 部署 → 新增部署 → 類型選「網頁應用程式」
執行身分：我　　存取權：所有人（含匿名）
5
複製部署後的網址，貼到上方「Apps Script Web App 網址」欄位，按測試連線確認成功後，按同步到 Sheets
6
下次接手的團長只需填入網址，按「從 Sheets 載入」就能取回所有資料 ✓
const SS_ID = '你的試算表ID'; // ← 換成你的ID const SUPPLY_SHEET = '耗材庫存'; const LOG_SHEET = '出入庫記錄'; const PHOTO_FOLDER = '耗材照片'; function doGet(e) { const action = e.parameter.action; if (action === 'ping') return ok({ status: 'ok' }); if (action === 'getAll') return ok(getAllData()); return ok({ error: 'unknown action' }); } function doPost(e) { try { const body = JSON.parse(e.postData.contents); const action = body.action; if (action === 'addSupply') return ok(addSupply(body.supply)); if (action === 'updateSupply') return ok(updateSupply(body.supply)); if (action === 'deleteSupply') return ok(deleteSupply(body.id)); if (action === 'addLog') return ok(addLog(body.log)); if (action === 'uploadPhoto') return ok(uploadPhoto(body.id, body.photo)); if (action === 'saveAll') return ok(saveAll(body.data)); return ok({ error: 'unknown action' }); } catch(err) { return ok({ error: err.message }); } } // ── 照片上傳：base64 → Drive → 公開縮圖 URL ── function uploadPhoto(supplyId, base64DataUrl) { try { const parts = base64DataUrl.split(','); const mime = parts[0].split(':')[1].split(';')[0]; // e.g. image/jpeg const ext = mime.split('/')[1]; const blob = Utilities.newBlob( Utilities.base64Decode(parts[1]), mime, 'supply_' + supplyId + '.' + ext ); const folder = getPhotoFolder(); // 刪除同名舊檔 const old = folder.getFilesByName(blob.getName()); while (old.hasNext()) old.next().setTrashed(true); // 上傳並公開 const file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); return { url: 'https://drive.google.com/thumbnail?id=' + file.getId() + '&sz=w400' }; } catch(err) { return { error: err.message }; } } function getPhotoFolder() { const folders = DriveApp.getFoldersByName(PHOTO_FOLDER); if (folders.hasNext()) return folders.next(); return DriveApp.createFolder(PHOTO_FOLDER); } // ── 批次存檔（saveAll） ── function saveAll(data) { const ss = SpreadsheetApp.openById(SS_ID); if (data.supplies) jsonToSheet(ss.getSheetByName(SUPPLY_SHEET), data.supplies); if (data.logs) jsonToSheet(ss.getSheetByName(LOG_SHEET), data.logs); return { saved: true }; } // ── 單筆 CRUD ── function getAllData() { const ss = SpreadsheetApp.openById(SS_ID); return { supplies: sheetToJson(ss.getSheetByName(SUPPLY_SHEET)), logs: sheetToJson(ss.getSheetByName(LOG_SHEET)) }; } function addSupply(s) { const sheet = SpreadsheetApp.openById(SS_ID).getSheetByName(SUPPLY_SHEET); ensureHeaders(sheet, ['id','name','cat','stock','unit','min','location','notes','photo']); sheet.appendRow([s.id,s.name,s.cat,s.stock,s.unit,s.min,s.location,s.notes,s.photo||'']); return { success: true }; } function updateSupply(s) { const sheet = SpreadsheetApp.openById(SS_ID).getSheetByName(SUPPLY_SHEET); const data = sheet.getDataRange().getValues(); const hdr = data[0]; const idCol = hdr.indexOf('id'); for (let i = 1; i < data.length; i++) { if (String(data[i][idCol]) === String(s.id)) { hdr.forEach(function(h, c) { if (s[h] !== undefined) sheet.getRange(i+1, c+1).setValue(s[h]); }); return { success: true }; } } return { error: '找不到 id: ' + s.id }; } function deleteSupply(id) { const sheet = SpreadsheetApp.openById(SS_ID).getSheetByName(SUPPLY_SHEET); const data = sheet.getDataRange().getValues(); const idCol = data[0].indexOf('id'); for (let i = 1; i < data.length; i++) { if (String(data[i][idCol]) === String(id)) { sheet.deleteRow(i + 1); return { success: true }; } } return { error: '找不到 id: ' + id }; } function addLog(log) { const sheet = SpreadsheetApp.openById(SS_ID).getSheetByName(LOG_SHEET); ensureHeaders(sheet, ['date','name','type','qty','unit','person','note']); sheet.appendRow([log.date,log.name,log.type,log.qty,log.unit,log.person,log.note||'']); return { success: true }; } function ensureHeaders(sheet, headers) { if (!sheet) return; if (sheet.getLastRow() === 0) sheet.appendRow(headers); } function sheetToJson(sheet) { if (!sheet) return []; const vals = sheet.getDataRange().getValues(); if (vals.length < 2) return []; const hdr = vals[0]; return vals.slice(1).map(function(r) { return Object.fromEntries(hdr.map(function(h, i) { return [h, r[i]]; })); }); } function jsonToSheet(sheet, arr) { if (!sheet || !arr || !arr.length) return; const hdr = Object.keys(arr[0]); sheet.clearContents(); sheet.appendRow(hdr); arr.forEach(function(r) { sheet.appendRow(hdr.map(function(h) { return r[h] !== undefined ? r[h] : ''; })); }); } function ok(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
